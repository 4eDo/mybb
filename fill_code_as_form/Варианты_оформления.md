# Сделать подставляемый в описании полей код кликабельным
Необходимо дополнить поле скрипта данными:
```
<script>
	const COLOR_INPUT_TEXT_fcaf = "";
	const NEED_HIDE_NAVLINKS = false;
	const TEMPLATE_SRC = "https://forumstatic.ru/files/001c/02/df/45236.html";

	document.addEventListener('click', function(event) {
	    let targetForm = event.target.closest('#targetForm');
	    if (targetForm && event.target.tagName === 'CODE') {
		const codeElement = event.target;
		const codeText = codeElement.textContent || codeElement.innerText;
	
		navigator.clipboard.writeText(codeText)
		    .then(() => {codeElement.classList.add('copied');setTimeout(() => {codeElement.classList.remove('copied');}, 1500); })
		    .catch(err => {console.error('Не удалось скопировать код в буфер обмена:', err);});
	    }
	});
</script>
```
